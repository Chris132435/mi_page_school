\section*{Geometría plana}

\subsection*{Triángulos}

\begin{tikzpicture}[scale=0.5, baseline=-0.5cm]
\draw (0,0) -- (3,0) -- (1,2) -- cycle;
\draw (0,0) node[below left] {$A$};
\draw (3,0) node[below right] {$B$};
\draw (1,2) node[above] {$C$};
\draw[<->] (0,-0.3) -- (3,-0.3) node[midway,below] {$c$};
\draw[<->] (3.2,0) -- (1.2,2) node[midway,right] {$a$};
\draw[<->] (-0.2,0) -- (0.8,2) node[midway,left] {$b$};
\end{tikzpicture}

\begin{itemize}[leftmargin=*]
    \item Suma de ángulos: $A+B+C = 180^\circ$
    \item Teorema de Pitágoras (rectángulo): $a^2+b^2=c^2$
    \item Área: $A = \frac{1}{2} \cdot \text{base} \cdot \text{altura}$
    \item Ley de senos: $\frac{a}{\sin A} = \frac{b}{\sin B} = \frac{c}{\sin C}$
    \item Ley de cosenos: $c^2 = a^2+b^2-2ab\cos C$
\end{itemize}

\begin{itemize}[leftmargin=*]
    \item Cuadrado: $P=4a$, $A=a^2$
    \item Rectángulo: $P=2(b+h)$, $A=bh$
    \item Paralelogramo: $A = b\cdot h$
    \item Trapecio: $A = \frac{(B+b)h}{2}$
    \item Rombo: $A = \frac{D\cdot d}{2}$
\end{itemize}

\subsection*{Círculo y circunferencia}

\begin{tikzpicture}[scale=0.5, baseline=-0.5cm]

\begin{itemize}[leftmargin=*]
    \item Longitud (perímetro): $L = 2\pi r$
    \item Área del círculo: $A = \pi r^2$
    \item Longitud de arco: $L_{\text{arco}} = \frac{\theta}{360^\circ} \cdot 2\pi r$
    \item Área del sector: $A_{\text{sector}} = \frac{\theta}{360^\circ} \cdot \pi r^2$
\end{itemize}

\section*{Geometría analítica (plano)}

\subsection*{Puntos y rectas}

\begin{itemize}[leftmargin=*]
    \item Distancia entre dos puntos: $d = \sqrt{(x_2-x_1)^2+(y_2-y_1)^2}$
    \item Punto medio: $M = \left(\frac{x_1+x_2}{2},\frac{y_1+y_2}{2}\right)$
    \item Pendiente: $m = \frac{y_2-y_1}{x_2-x_1}$
    \item Ecuación de la recta: $y-y_1 = m(x-x_1)$
    \item Distancia de un punto a una recta $Ax+By+C=0$: $d = \frac{|Ax_0+By_0+C|}{\sqrt{A^2+B^2}}$
\end{itemize}

\subsection*{Elipse}
\[
\frac{(x-h)^2}{a^2} + \frac{(y-k)^2}{b^2} = 1
\]

\subsection*{Hipérbola}
\[
\frac{(x-h)^2}{a^2} - \frac{(y-k)^2}{b^2} = 1
