\section*{Cinemática}

\subsection*{Movimiento en línea recta con \(a\) cte}
\begin{align*}
    v &= v_0 + a t \\
    x &= x_0 + v_0 t + \tfrac{1}{2} a t^2 \\
    v^2 &= v_0^2 + 2a (x - x_0)
\end{align*}

\subsection*{MRU (distancia)}
\[
x = x_0 + v t
\]

\subsection*{Caída libre desde altura \(h\)}
\begin{align*}
    v &= g t \\
    h &= \tfrac{1}{2} g t^2 \\
    v^2 &= 2 g h
\end{align*}

\subsection*{Lanzamiento vertical hacia arriba}
\begin{align*}
    y &= y_0 + v_0 t - \tfrac{1}{2} g t^2 \\
    v &= v_0 - g t \\
    v^2 &= v_0^2 - 2g (y - y_0)
\end{align*}

\subsection*{Tiro parabólico completo}
\begin{align*}
    R &= \frac{v_0^2 \sen(2\theta)}{g} \\
    H &= \frac{v_0^2 \sen^2\theta}{2g} \\
    T &= \frac{2 v_0 \sen\theta}{g}
\end{align*}

\subsection*{Movimiento horizontal de proyectiles}
\begin{align*}
    x &= v_0 t \\
    y &= \tfrac{1}{2} g t^2
\end{align*}

\subsection*{Plano inclinado (sin fricción)}
\begin{align*}
    a &= g \sen\theta \\
    v^2 &= 2 g \sen\theta \, d
\end{align*}

\subsection*{Movimiento circular uniforme}
\begin{align*}
    v &= \omega R \\
    a_c &= \frac{v^2}{R} = \omega^2 R \\
    T &= \frac{2\pi}{\omega},\quad f = \frac{1}{T}
\end{align*}

\subsection*{Fuerza centrípeta}
\[
F_c = m \frac{v^2}{R} = m \omega^2 R
\]

\subsection*{Aceleración angular constante}
\begin{align*}
    \omega &= \omega_0 + \alpha t \\
    \theta &= \theta_0 + \omega_0 t + \tfrac{1}{2} \alpha t^2 \\
    \omega^2 &= \omega_0^2 + 2\alpha (\theta - \theta_0)
\end{align*}

\subsection*{Aceleración tangencial y centrípeta}
\[
\vec{a} = \vec{a}_t + \vec{a}_c,\quad a_t = \alpha R,\quad a_c = \omega^2 R
\]

\section*{Dinámica}

\subsection*{Leyes de Newton}
\begin{align*}
    \text{1ª: } & \sum \vec{F}=0 \Rightarrow \vec{v}=cte \\
    \text{2ª: } & \sum \vec{F}=m\vec{a} \\
    \text{3ª: } & \vec{F}_{12}=-\vec{F}_{21}
\end{align*}

\subsection*{Fricción}
\[
f_k = \mu_k N,\qquad f_s^{\max} = \mu_s N
\]

\subsection*{Ley de Hooke}
\[
F = -k x
\]

\subsection*{Ley de gravitación universal}
\[
F_g = G \frac{m_1 m_2}{r^2},\quad G = 6.67\times10^{-11}
\]

\subsection*{Aceleración de gravedad}
\[
g = \frac{G M}{R^2}
\]

\subsection*{Peso}
\[
\vec{P} = m \vec{g}
\]

\section*{Trabajo, Energía y Potencia}

\subsection*{Trabajo mecánico}
\[
W = \vec{F} \cdot \Delta \vec{r} = F \Delta r \cos\theta
\]

\subsection*{Potencia}
\[
P = \frac{W}{\Delta t} = \vec{F} \cdot \vec{v}
\]

\subsection*{Energía cinética}
\[
K = \tfrac{1}{2} m v^2
\]

\subsection*{Energía cinética de rotación}
\[
K_{\text{rot}} = \tfrac{1}{2} I \omega^2
\]

\subsection*{Teorema trabajo–energía}
\[
W_{\text{neto}} = \Delta K
\]

\subsection*{Energía potencial gravitatoria}
\[
U = m g h
\]

\subsection*{Energía potencial elástica}
\[
U_e = \tfrac{1}{2} k x^2
\]

\subsection*{Conservación de la energía mecánica}
\[
E_m = K + U = \text{constante}
\]

\section*{Cantidad de Movimiento y Colisiones}

\subsection*{Momento lineal}
\[
\vec{p} = m \vec{v}
\]

\subsection*{Conservación del momento lineal}
\[
\sum \vec{p}_{\text{inicial}} = \sum \vec{p}_{\text{final}}
\]

\subsection*{Colisión elástica (1D)}
\begin{align*}
    m_1 v_{1i} + m_2 v_{2i} &= m_1 v_{1f} + m_2 v_{2f} \\
    \tfrac{1}{2}m_1 v_{1i}^2 + \tfrac{1}{2}m_2 v_{2i}^2 &= \tfrac{1}{2}m_1 v_{1f}^2 + \tfrac{1}{2}m_2 v_{2f}^2
\end{align*}

\subsection*{Colisión perfectamente inelástica}
\[
m_1 v_{1i} + m_2 v_{2i} = (m_1+m_2) V_f
\]

\section*{Dinámica de Rotación}

\subsection*{Momentos de inercia}
\begin{align*}
    \text{Anillo: } & I = M R^2 \\
    \text{Disco: } & I = \tfrac{1}{2} M R^2 \\
    \text{Barra (centro): } & I = \tfrac{1}{12} M L^2 \\
    \text{Esfera sólida: } & I = \tfrac{2}{5} M R^2
\end{align*}

\subsection*{Momento angular}
\[
\vec{L} = I \vec{\omega}
\]

\subsection*{Torque}
\[
\vec{\tau} = \vec{r} \times \vec{F},\qquad \tau = I \alpha
\]

\subsection*{Conservación del momento angular}
\[
\sum \vec{L}_{\text{inicial}} = \sum \vec{L}_{\text{final}}
\]

\subsection*{Precesión}
\[
\Omega_p = \frac{\tau}{L}
\]

\section*{Fluidos}

\subsection*{Densidad}
\[
\rho = \frac{m}{V}
\]

\subsection*{Presión}
\[
P = \frac{F}{A}
\]

\subsection*{Principio de Pascal}
\[
\frac{F_1}{A_1} = \frac{F_2}{A_2} \quad\Rightarrow\quad F_2 = F_1 \frac{A_2}{A_1}
\]
