\section*{Cinemática}

\subsection*{Movimiento en línea recta con aceleración constante}
\begin{align*}
    v &= v_0 + a t \\
    x &= x_0 + v_0 t + \frac{1}{2} a t^2 \\
    v^2 &= v_0^2 + 2a (x - x_0)
\end{align*}

\subsection*{Función de distancia (MRU)}
\[
x = x_0 + v t
\]

\subsection*{Caída libre desde altura \(h\)}
\begin{align*}
    v &= g t \\
    h &= \frac{1}{2} g t^2 \\
    v^2 &= 2 g h
\end{align*}

\subsection*{Movimiento vertical de proyectiles (lanzamiento hacia arriba)}
\begin{align*}
    y &= y_0 + v_0 t - \frac{1}{2} g t^2 \\
    v &= v_0 - g t \\
    v^2 &= v_0^2 - 2g (y - y_0)
\end{align*}

\subsection*{Tiro parabólico (completo)}
\begin{align*}
    \text{Alcance: } R &= \frac{v_0^2 \sen(2\theta)}{g} \\
    \text{Altura máxima: } H &= \frac{v_0^2 \sen^2\theta}{2g} \\
    \text{Tiempo de vuelo: } T &= \frac{2 v_0 \sen\theta}{g}
\end{align*}

\subsection*{Movimiento horizontal de proyectiles}
\begin{align*}
    x &= v_0 t \\
    y &= \frac{1}{2} g t^2
\end{align*}

\subsection*{Movimiento en plano inclinado (sin fricción)}
\begin{align*}
    a &= g \sen\theta \\
    v^2 &= 2 g \sen\theta \, d \quad (\text{partiendo del reposo})
\end{align*}

\subsection*{Movimiento circular uniforme}
\begin{align*}
    v &= \omega R \\
    a_c &= \frac{v^2}{R} = \omega^2 R \\
    T &= \frac{2\pi}{\omega}, \quad f = \frac{1}{T}
\end{align*}

\subsection*{Fuerza centrípeta}
\[
F_c = m a_c = m \frac{v^2}{R} = m \omega^2 R
\]

\subsection*{Aceleración constante en movimiento circular (aceleración angular constante)}
\begin{align*}
    \omega &= \omega_0 + \alpha t \\
    \theta &= \theta_0 + \omega_0 t + \frac{1}{2} \alpha t^2 \\
    \omega^2 &= \omega_0^2 + 2\alpha (\theta - \theta_0)
\end{align*}

\subsection*{Aceleración circular (tangencial y centrípeta)}
\[
\vec{a} = \vec{a}_t + \vec{a}_c, \quad a_t = \alpha R, \quad a_c = \omega^2 R
\]

\section*{Dinámica}

\subsection*{Leyes de Newton}
\begin{align*}
    \text{1ª Ley:} & \quad \sum \vec{F} = 0 \Rightarrow \vec{v} = \text{constante} \\
    \text{2ª Ley:} & \quad \sum \vec{F} = m \vec{a} \\
    \text{3ª Ley:} & \quad \vec{F}_{12} = -\vec{F}_{21}
\end{align*}

\subsection*{Fuerza de fricción (cinética y estática)}
\[
f_k = \mu_k N, \qquad f_s^{\max} = \mu_s N
\]

\subsection*{Ley de Hooke (fuerza elástica)}
\[
F = -k x
\]

\subsection*{Fuerza gravitacional (atracción universal)}
\[
F_g = G \frac{m_1 m_2}{r^2}, \quad G = 6.67\times10^{-11}\,\text{N·m}^2/\text{kg}^2
\]

\subsection*{Aceleración de gravedad (en la superficie de un planeta)}
\[
g = \frac{G M}{R^2}
\]

\subsection*{Peso (fuerza de gravedad)}
\[
\vec{P} = m \vec{g}
\]

\section*{Trabajo, Energía y Potencia}

\subsection*{Trabajo mecánico}
\[
W = \vec{F} \cdot \Delta \vec{r} = F \Delta r \cos\theta
\]

\subsection*{Potencia}
\[
P = \frac{W}{\Delta t} = \vec{F} \cdot \vec{v}
\]

\subsection*{Energía cinética}
\[
K = \frac{1}{2} m v^2
\]

\subsection*{Energía cinética de un cuerpo en rotación masiva}
\[
K_{\text{rot}} = \frac{1}{2} I \omega^2
\]

\subsection*{Principio trabajo – energía}
\[
W_{\text{neto}} = \Delta K = K_f - K_i
\]

\subsection*{Energía potencial gravitatoria (cerca de la superficie)}
\[
U = m g h
\]

\subsection*{Energía potencial elástica}
\[
U_e = \frac{1}{2} k x^2
\]

\subsection*{Conservación de la energía mecánica (sin fuerzas no conservativas)}
\[
E_m = K + U = \text{constante}
\]

\section*{Cantidad de Movimiento y Colisiones}

\subsection*{Momento lineal (cantidad de movimiento)}
\[
\vec{p} = m \vec{v}
\]

\subsection*{Ley de conservación del momento lineal}
\[
\sum \vec{p}_{\text{inicial}} = \sum \vec{p}_{\text{final}}
\]

\subsection*{Colisión elástica (conservación de energía cinética y momento)}
\[
m_1 v_{1i} + m_2 v_{2i} = m_1 v_{1f} + m_2 v_{2f}
\]
\[
\frac{1}{2}m_1 v_{1i}^2 + \frac{1}{2}m_2 v_{2i}^2 = \frac{1}{2}m_1 v_{1f}^2 + \frac{1}{2}m_2 v_{2f}^2
\]

\subsection*{Colisión inelástica (solo momento, se pierde energía)}
\[
m_1 v_{1i} + m_2 v_{2i} = (m_1 + m_2) V_f \quad (\text{perfectamente inelástica})
\]

\section*{Dinámica de Rotación}

\subsection*{Momento de inercia (casos comunes)}
\begin{align*}
    \text{Anillo o aro: } & I = M R^2 \\
    \text{Disco sólido: } & I = \frac{1}{2} M R^2 \\
    \text{Barra delgada (centro): } & I = \frac{1}{12} M L^2 \\
    \text{Esfera sólida: } & I = \frac{2}{5} M R^2
\end{align*}

\subsection*{Momento angular}
\[
\vec{L} = I \vec{\omega}
\]

\subsection*{Torque o momento de fuerza}
\[
\vec{\tau} = \vec{r} \times \vec{F}, \qquad \tau = I \alpha
\]

\subsection*{Conservación del momento angular}
\[
\sum \vec{L}_{\text{inicial}} = \sum \vec{L}_{\text{final}}
\]

\subsection*{Aplicación del momento angular (precesión, giroscopios)}
\[
\Omega_p = \frac{\tau}{L} \quad (\text{frecuencia de precesión})
\]

\section*{Fluidos}

\subsection*{Densidad}
\[
\rho = \frac{m}{V}
\]

\subsection*{Presión}
\[
P = \frac{F}{A}
\]

\subsection*{Principio de Pascal}
\[
\Delta P = \frac{F_1}{A_1} = \frac{F_2}{A_2} \quad \Rightarrow \quad F_2 = F_1 \frac{A_2}{A_1}
\]
